{{ if .IsHome }} {{- else if (or .IsPage .IsSection) }} {{- /* BreadcrumbList */ -}} {{- $url := replace .Parent.Permalink (printf "%s" site.Home.Permalink) "" -}} {{- $lang_url := strings.TrimPrefix (printf "%s/" .Lang) $url -}} {{- $bc_list := split $lang_url "/" -}} {{- $crumbs := slice -}} {{- range $index, $element := $bc_list -}} {{- if and (gt (len $element) 0) (ne $element "") -}} {{- $path := printf "%s/" $element -}} {{- $bc_pg := site.GetPage $path -}} {{- if $bc_pg -}} {{- $crumbs = $crumbs | append (dict "name" $bc_pg.Name "item" $bc_pg.Permalink) -}} {{- end -}} {{- end -}} {{- end -}} {{- $crumbs = $crumbs | append (dict "name" .Name "item" .Permalink) -}} {{- if .IsPage }} {{- end }} {{- end -}}
